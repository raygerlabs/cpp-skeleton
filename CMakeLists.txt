#---------------------------------------------------------------------#
cmake_minimum_required(VERSION 3.20)
project(project-init-cpp VERSION 0.1)
#---------------------------------------------------------------------#
if (NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Release)
endif()
option(BUILD_SHARED_LIBS "Build shared libraries" ON)
option(BUILD_TESTING "Build unit tests" ON)
#---------------------------------------------------------------------#
list(APPEND CMAKE_PREFIX_PATH ${CMAKE_BINARY_DIR})
list(APPEND CMAKE_MODULE_PATH ${CMAKE_BINARY_DIR})
#---------------------------------------------------------------------#
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
#---------------------------------------------------------------------#
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)
set(CMAKE_CXX_VISIBILITY_PRESET hidden)
set(CMAKE_VISIBILITY_INLINES_HIDDEN ON)
set_property(GLOBAL PROPERTY USE_FOLDERS ON)
set(CMAKE_SUPPRESS_REGENERATION true)
#---------------------------------------------------------------------#
add_subdirectory(src)
include(CTest)
if (BUILD_TESTING)
  include(GoogleTest)
  add_subdirectory(tests)
endif()
#---------------------------------------------------------------------#
file(WRITE ${CMAKE_BINARY_DIR}/VERSION ${PROJECT_VERSION})
set(CPACK_PACKAGE_VERSION "${CMAKE_BINARY_DIR}/VERSION")
install(FILES ${CPACK_PACKAGE_VERSION} DESTINATION .)
set(CPACK_RESOURCE_FILE_README ${CMAKE_SOURCE_DIR}/README.md)
set(CPACK_RESOURCE_FILE_LICENSE ${CMAKE_SOURCE_DIR}/LICENSE)
install(FILES ${CPACK_RESOURCE_FILE_README} DESTINATION .)
install(FILES ${CPACK_RESOURCE_FILE_LICENSE} DESTINATION .)
#---------------------------------------------------------------------#
set(ARCHIVE_NAME "${PROJECT_NAME}")
set(ARCHIVE_NAME "${ARCHIVE_NAME}-v${PROJECT_VERSION}")
set(ARCHIVE_NAME "${ARCHIVE_NAME}-${CMAKE_SYSTEM_NAME}")
set(ARCHIVE_NAME "${ARCHIVE_NAME}-${CMAKE_SYSTEM_PROCESSOR}")
string(TOLOWER "${ARCHIVE_NAME}" ARCHIVE_NAME)
set(CPACK_PACKAGE_FILE_NAME "${ARCHIVE_NAME}")
set(CPACK_SOURCE_PACKAGE_FILE_NAME "${ARCHIVE_NAME}-src")
set(CPACK_OUTPUT_FILE_PREFIX ${CMAKE_BINARY_DIR})
set(CPACK_PACKAGE_DIRECTORY ${CMAKE_BINARY_DIR})
set(ARCHIVE_TYPE TGZ)
set(ARCHIVE_EXTENSION ".tar.gz")
if(WIN32)
  set(ARCHIVE_TYPE ZIP)
  set(ARCHIVE_EXTENSION ".zip")
endif()
set(CPACK_GENERATOR ${ARCHIVE_TYPE})
set(CPACK_SOURCE_GENERATOR ${ARCHIVE_TYPE})
set(ARCHIVE_NAME "${ARCHIVE_NAME}${ARCHIVE_EXTENSION}")
file(WRITE "${CMAKE_BINARY_DIR}/archive_name.txt" "${ARCHIVE_NAME}")
include(CPack)
#---------------------------------------------------------------------#
